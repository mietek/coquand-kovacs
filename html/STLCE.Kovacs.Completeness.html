-- (_≈_)
mutual
  infix 3 _‼≫_
  _‼≫_ : ∀ {A Γ} → Γ ⊢ A → Γ ‼⊩ A → Set
  _‼≫_ {⎵}      {Γ} M N = M ∼ embⁿᶠ N
  _‼≫_ {A ⇒ B} {Γ} M f = ∀ {Γ′} → (η : Γ′ ⊇ Γ) {N : Γ′ ⊢ A} {a : Γ′ ⊩ A}
                                    (p : N ≫ a)
                                 → ren η M ∙ N ≫ f η a
  _‼≫_ {A ⩕ B}  {Γ} M s = π₁ M ≫ proj₁ s × π₂ M ≫ proj₂ s
  _‼≫_ {⫪}     {Γ} M s = ⊤
  _‼≫_ {⫫}     {Γ} M s = elim⊥ s
  _‼≫_ {A ⩖ B}  {Γ} M s = M ⁇ ι₁ (` zero) ∥ ι₂ (` zero) ≫
                          elim⊎ s (λ a → return {A ⩖ B} (inj₁ a))
                                  (λ b → return {A ⩖ B} (inj₂ b))

  infix 3 _≫_
  postulate
    _≫_ : ∀ {A Γ} → Γ ⊢ A → Γ ⊩ A → Set

  -- M ≫ k = {!!}


-- (_≈ᶜ_)
infix 3 _≫⋆_
data _≫⋆_ : ∀ {Γ Ξ} → Γ ⊢⋆ Ξ → Γ ⊩⋆ Ξ → Set
  where
    ∅   : ∀ {Γ} → ∅ {Γ = Γ} ≫⋆ ∅

    _,_ : ∀ {Γ Ξ A} → {σ : Γ ⊢⋆ Ξ} {ρ : Γ ⊩⋆ Ξ} {M : Γ ⊢ A} {a : Γ ⊩ A}
                    → (χ : σ ≫⋆ ρ) (p : M ≫ a)
                    → σ , M ≫⋆ ρ , a

postulate
  id≫⋆ : ∀ {Γ} → idₛ {Γ} ≫⋆ idᵥ

-- id≫⋆ {∅}     = ∅
-- id≫⋆ {Γ , A} = id≫⋆ ⬖≫ wkₑ idₑ , reflect≫ (` zero)


complete : ∀ {Γ A} → (M : Γ ⊢ A)
                   → M ∼ embⁿᶠ (nf M)
complete M = coe ((_∼ embⁿᶠ (reify (eval idᵥ M))) & idsub M)
                 (reify≫ (eval≫ id≫⋆ M))
