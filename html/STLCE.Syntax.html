module STLCE.Syntax where

open import Prelude public


--------------------------------------------------------------------------------


infix  9 _⩕_
infix  8 _⩖_
infixr 7 _⇒_
data 𝒯 : Set
  where
    ⎵    : 𝒯

    _⇒_ : (A B : 𝒯) → 𝒯

    _⩕_  : (A B : 𝒯) → 𝒯

    ⫪   : 𝒯

    ⫫   : 𝒯

    _⩖_  : (A B : 𝒯) → 𝒯


data 𝒞 : Set
  where
    ∅   : 𝒞

    _,_ : (Γ : 𝒞) (A : 𝒯) → 𝒞


infix 4 _∋_
data _∋_ : 𝒞 → 𝒯 → Set
  where
    zero : ∀ {Γ A} → Γ , A ∋ A

    suc  : ∀ {Γ A B} → (i : Γ ∋ A)
                     → Γ , B ∋ A


infix 3 _⊢_
data _⊢_ : 𝒞 → 𝒯 → Set
  where
    `     : ∀ {Γ A} → (i : Γ ∋ A)
                    → Γ ⊢ A

    ƛ     : ∀ {Γ A B} → (M : Γ , A ⊢ B)
                      → Γ ⊢ A ⇒ B

    _∙_   : ∀ {Γ A B} → (M : Γ ⊢ A ⇒ B) (N : Γ ⊢ A)
                      → Γ ⊢ B

    _,_   : ∀ {Γ A B} → (M : Γ ⊢ A) (N : Γ ⊢ B)
                      → Γ ⊢ A ⩕ B

    π₁    : ∀ {Γ A B} → (M : Γ ⊢ A ⩕ B)
                      → Γ ⊢ A

    π₂    : ∀ {Γ A B} → (M : Γ ⊢ A ⩕ B)
                      → Γ ⊢ B

    τ     : ∀ {Γ} → Γ ⊢ ⫪

    φ     : ∀ {Γ C} → (M : Γ ⊢ ⫫)
                    → Γ ⊢ C

    ι₁    : ∀ {Γ A B} → (M : Γ ⊢ A)
                      → Γ ⊢ A ⩖ B

    ι₂    : ∀ {Γ A B} → (M : Γ ⊢ B)
                      → Γ ⊢ A ⩖ B

    _⁇_∥_ : ∀ {Γ A B C} → (M : Γ ⊢ A ⩖ B) (N₁ : Γ , A ⊢ C) (N₂ : Γ , B ⊢ C)
                        → Γ ⊢ C
